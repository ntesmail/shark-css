使用方式

scss文件中引用node_modules下的@ntesmail/shark-css

npm install @ntesmail/shark-css --save

// 皮肤 optional
@import '@ntesmail/shark-css/src/main/webapp/style/scss/shark-bootstrap/bootstrap-salte/variables';
// 主文件
@import '@ntesmail/shark-css/src/main/webapp/style/scss/shark-bootstrap/shark-bootstrap';
// 皮肤 optional
@import '@ntesmail/shark-css/src/main/webapp/style/scss/shark-bootstrap/bootstrap-salte/bootswatch';
// 布局
@import '@ntesmail/shark-css/src/main/webapp/style/scss/layout/layout';
